		var LoginController = [ "$scope", "$timeout", "AccountApi", "SystemApi", "SisenseApi", function( $scope, $timeout, accountApi, systemApi, sisenseApi ) {
			$scope.resetPasswordModalState = {};
			sisenseApi.Init( 'https://sisense-prod.skuvault.com' );

			$scope.init = function() {
				systemApi.GetSystemMessage()
					.onData( function( data ) {
						$scope.systemMessage = data;
					} );
				sisenseApi.LogoutSilently();
			};

			$scope.login = function( event ) {
				event.preventDefault();
				sisenseApi.LogoutSilently().finally( () => {
					$( 'form[name="loginForm"]' ).submit();
				} );
			}

			$scope.showResetPassword = function() {
				$scope.resetPasswordModalState = { Shown: true };
			};

			$scope.resetPassword = function( email ) {
				$scope.forgotPasswordAlertInfo = null;
				accountApi.SendResetPasswordEmail( email )
					.onData( function( data ) {
						$scope.resetPasswordModalState.Shown = false;
						$scope.forgotPasswordAlertInfo = { title: "Forgot Password", messages: [ data ], type: "success" };
					} )
					.onBlank( function( data ) {
						$scope.resetPasswordModalState.Shown = false;
						$scope.forgotPasswordAlertInfo = { title: "Forgot Password", messages: data, type: "error" };
					} )
					.withState( $scope.resetPasswordModalState );
			};
		} ];
	</script>
